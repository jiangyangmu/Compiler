int main()
{
    char i8;
    unsigned char u8;
    short i16;
    unsigned short u16;
    int i32;
    unsigned int u32;
    long i64;
    unsigned long u64;

    int result;

    // unsigned long int, * => unsigned long int
    result = u64 + i8;
    result = u64 + i16;
    result = u64 + i32;
    result = u64 + i64;
    result = u64 + u8;
    result = u64 + u16;
    result = u64 + u32;

    // long int, * => long int
    result = i64 + i8;
    result = i64 + i16;
    result = i64 + i32;
    result = i64 + u8;
    result = i64 + u16;
    result = i64 + u32;

    // unsigned int, * => unsigned int
    result = u32 + i8;
    result = u32 + i16;
    result = u32 + i32;
    result = u32 + u8;
    result = u32 + u16;

    // int, * => int
    result = i32 + i8;
    result = i32 + i16;
    result = i32 + u8;
    result = i32 + u16;

    return 0;
}
